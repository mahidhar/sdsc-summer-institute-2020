# Start Spark session

import pyspark
from pyspark.sql import SparkSession

conf = pyspark.SparkConf().setAll([('spark.master', 'local[*]'),
                                   ('spark.app.name', 'PySpark Data Exploration')])
spark = SparkSession.builder.config(conf=conf).getOrCreate()
